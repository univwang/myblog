class Solution {
public:
    const int p = 1e9 + 7;
    int distinctSubseqII(string s) {
        vector<int> dp(26);
        int sum = 0;
        for(int i = 0; i < s.size(); i ++ ) {
            sum %= p;
            int tmp = dp[s[i] - 'a']; 
            dp[s[i] - 'a'] = sum + 1;
            sum += (dp[s[i] - 'a'] - tmp + p) % p; 
        }
        int ans = 0;
        for(int i = 0; i < 26; i ++) {
            ans += dp[i];
            ans %= p;
        }
        return ans;
    }
};
